+import PropTypes from 'prop-types';
+import { useState, useEffect, useCallback } from 'react';
+
+/**
+ * NextRoundTimer component - Displays a 3-second countdown before the next round starts
+ * Shows a visual indicator (shrinking bar or numeric countdown)
+ */
+export function NextRoundTimer({
+  isVisible,
+  onComplete,
+  nextRoundNumber
+}) {
+  const [timeRemaining, setTimeRemaining] = useState(3);
+  const [progress, setProgress] = useState(100);
+
+  const startTimer = useCallback(() => {
+    setTimeRemaining(3);
+    setProgress(100);
+    
+    const timer = setInterval(() => {
+      setTimeRemaining(prev => {
+        const newTime = prev - 1;
+        const newProgress = (newTime / 3) * 100;
+        setProgress(newProgress);
+        
+        if (newTime <= 0) {
+          clearInterval(timer);
+          if (onComplete) {
+            setTimeout(onComplete, 0);
+          }
+          return 0;
+        }
+        return newTime;
+      });
+    }, 1000);
+    
+    return () => clearInterval(timer);
+  }, [onComplete]);
+
+  useEffect(() => {
+    if (isVisible) {
+      const cleanup = startTimer();
+      return cleanup;
+    }
+  }, [isVisible, startTimer]);
+
+  if (!isVisible) return null;
+
+  // Calculate the width of the progress bar based on remaining time
+  const barWidth = `${progress}%`;
+
+  return (
+    <div className="next-round-timer" role="timer" aria-live="polite">
+      <div className="timer-container">
+        <h3 className="timer-title">Next Round Starting...</h3>
+        <p className="round-info">Round {nextRoundNumber}</p>
+        
+        <div className="countdown-bar-container">
+          <div 
+            className="countdown-bar"
+            style={{ width: barWidth }}
+            aria-label={`Countdown progress: ${Math.round(progress)}%`}
+          />
+        </div>
+        
+        <div className="numeric-countdown" aria-label={`Countdown: ${timeRemaining}`}>
+          {timeRemaining}
+        </div>
+      </div>
+    </div>
+  );
+}
+
+NextRoundTimer.propTypes = {
+  isVisible: PropTypes.bool.isRequired,
+  onComplete: PropTypes.func.isRequired,
+  nextRoundNumber: PropTypes.number.isRequired
+};
+
+export default NextRoundTimer;
